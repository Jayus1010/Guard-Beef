wsHandover.getRange(1, 1, 1, headersHandover.length).setValues([headersHandover]); } /** * Helper untuk meng-hash PIN menggunakan SHA-256 demi keamanan data * Menggunakan perulangan tradisional karena byte array dari GAS tidak memiliki metode .map() */ function hashPin(pin) { const digest = Utilities.computeDigest(Utilities.DigestAlgorithm.SHA_256, String(pin)); let hash = ''; for (let i = 0; i < digest.length; i++) { const byteVal = digest[i]; hash += ('0' + (byteVal & 0xFF).toString(16)).slice(-2); } return hash; } /** * Fungsi Wajib untuk menerima kiriman data (POST) dari GitHub Pages */ function doPost(e) { try { const requestData = JSON.parse(e.postData.contents); let result = {}; if (requestData.action === 'login') { result = login(requestData.username, requestData.pin); } else if (requestData.action === 'submitPatrol') { result = submitPatrol(requestData.data); } else if (requestData.action === 'getRecentLogs') { result = getRecentLogs(requestData.username); } else if (requestData.action === 'getAllLogs') { result = getAllLogs(); } else if (requestData.action === 'getGuards') { result = getGuards(); } else if (requestData.action === 'addGuard') { result = addGuard(requestData.username, requestData.pin, requestData.nama); } else if (requestData.action === 'deleteGuard') { result = deleteGuard(requestData.username); } else if (requestData.action === 'submitHandover') { result = submitHandover(requestData.data); } else if (requestData.action === 'getRecentHandovers') { result = getRecentHandovers(requestData.username); } else if (requestData.action === 'getAllHandovers') { result = getAllHandovers(); } else if (requestData.action === 'getPendingHandovers') { result = getPendingHandovers(requestData.username); } else if (requestData.action === 'approveHandover') { result = approveHandover(requestData.senderUsername, requestData.timestamp, requestData.approverUsername); } else { result = { success: false, message: 'Action tidak dikenali oleh server.' }; } return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ success: false, message: error.toString() })) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { if (e && e.parameter && e.parameter.setup === '1') { try { setupDatabase(); return ContentService.createTextOutput("Database Setup BERHASIL! Sheet 'Users' dan 'LogPatroli' telah dikonfigurasi dengan benar di Google Spreadsheet.") .setMimeType(ContentService.MimeType.TEXT); } catch (error) { return ContentService.createTextOutput("Gagal melakukan Setup Database: " + error.toString()) .setMimeType(ContentService.MimeType.TEXT); } } // Mengembalikan tampilan HTML (index.html) saat Web App dibuka langsung di browser const template = HtmlService.createTemplateFromFile('index'); try { template.webAppUrl = ScriptApp.getService().getUrl(); } catch (err) { template.webAppUrl = ''; } return template.evaluate() .setTitle('Guard Beef - Security') .addMetaTag('viewport', 'width=device-width, initial-scale=1') .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.DEFAULT); } // ========================================== // LOGIKA BISNIS // ========================================== function login(username, pin) { try { const ss = getSpreadsheet(); let ws = ss.getSheetByName('Users'); if(!ws) { setupDatabase(); ws = ss.getSheetByName('Users'); } const data = ws.getDataRange().getValues(); const hashedInputPin = hashPin(pin); for (let i = 1; i < data.length; i++) { const storedUsername = String(data[i][0]).trim(); const storedPin = String(data[i][1]).trim(); const role = data[i][3] ? String(data[i][3]).trim() : 'GUARD'; // Mendukung pencocokan hash SHA-256 atau fallback ke plain text jika belum di-hash const isPinMatch = (storedPin.length === 64) ? (storedPin === hashedInputPin) : (storedPin === String(pin)); if (storedUsername === String(username).trim() && isPinMatch) { return { success: true, nama: data[i][2], role: role, username: storedUsername }; } } return { success: false, message: 'Username atau PIN salah!' }; } catch (error) { return { success: false, message: 'Gagal login: ' + error.toString() }; } } function submitPatrol(data) { try { const { username, nama, qrCode, lat, lng, mapsUrl, kondisi, catatan } = data; return { success: true, message: `Patroli di area "${qrCode}" berhasil dicatat!` }; } catch (error) { return { success: false, message: 'Gagal mencatat patroli: ' + error.toString() }; } } function getRecentLogs(username) { try { const ss = getSpreadsheet(); let wsLog = ss.getSheetByName('LogPatroli'); if (!wsLog) { setupDatabase(); wsLog = ss.getSheetByName('LogPatroli'); } const data = wsLog.getDataRange().getValues(); const logs = []; // Ambil 10 log terakhir khusus untuk user yang bersangkutan for (let i = data.length - 1; } return { success: true, logs: logs }; } catch (error) { return { success: false, message: error.toString(), logs: [] }; } } function getGuards() { try { const ss = getSpreadsheet(); let ws = ss.getSheetByName('Users'); if (!ws) { setupDatabase(); ws = ss.getSheetByName('Users'); } const data = ws.getDataRange().getValues(); const guards = []; for (let i = 1; i < data.length; i++) { const role = data[i][3] ? String(data[i][3]).trim() : 'GUARD'; if (role === 'GUARD') { guards.push({ username: String(data[i][0]), nama: String(data[i][2]) }); } } return { success: true, guards: guards }; } catch (error) { return { success: false, message: error.toString(), guards: [] }; } } function addGuard(username, pin, nama) { try { const ss = getSpreadsheet(); let ws = ss.getSheetByName('Users'); if (!ws) { setupDatabase(); ws = ss.getSheetByName('Users'); } const data = ws.getDataRange().getValues(); } } return { success: true, handovers: pending }; } catch (error) { return { success: false, message: error.toString(), handovers: [] }; } } function approveHandover(senderUsername, timestamp, approverUsername) { try { const ss = getSpreadsheet(); ws.getRange(i + 1, 14).setValue(new Date()); return { success: true, message: 'Serah terima berhasil disetujui!' }; } } return { success: false, message: 'Laporan serah terima tidak ditemukan atau sudah disetujui sebelumnya.' };
